Classificação Forense de Vidros utilizando Redes Neurais Artificiais¶

Dataset:¶

Glass Identification, disponivel em https://archive.ics.uci.edu/dataset/42/glass+identification

Problema:¶

Na ciência forense (investigação criminal), fragmentos de vidro são frequentemente encontrados em cenas de crimes (roubos, atropelamentos). O problema é: como saber se o caco de vidro encontrado na roupa do suspeito veio da janela quebrada na cena do crime? O vidro possui uma "assinatura química" baseada nos elementos usados na sua fabricação (Sódio, Magnésio, Alumínio, etc.). A Rede Neural analisa essas 9 propriedades químicas e classifica a origem do vidro com base em características impossíveis de serem reconhecidas a olho nu.

Variáveis: temos uma propriedade física e 8 elementos químicos:

  • RI (Refractive Index): Índice de Refração (medida de quanto a luz "dobra" ao passar pelo vidro).

  • Na (Sodium): Sódio.

  • Mg (Magnesium): Magnésio.

  • Al (Aluminum): Alumínio.

  • Si (Silicon): Silício (o principal componente do vidro/areia).

  • K (Potassium): Potássio.

  • Ca (Calcium): Cálcio.

  • Ba (Barium): Bário.

  • Fe (Iron): Ferro.

Classes

Classe Categoria Descrição
1 Janelas de Construção (Float) Vidro padrão, moderno e plano (flutuado em estanho).
2 Janelas de Construção (Não-Float) Quimicamente similar à Classe 1, com variações de Magnésio.
3 Janelas de Veículo (Float) Vidros de carros (para-brisas ou laterais).
4 Janelas de Veículo (Não-Float) Classe Ausente (Não existe no dataset).
5 Containers (Recipientes) Garrafas, potes de conserva, frascos de perfume/remédio.
6 Louça de Mesa Copos de beber, taças, pratos de vidro.
7 Faróis Vidro protetor de faróis de carros.

def preparar_dados_geral():
    url = "https://archive.ics.uci.edu/ml/machine-learning-databases/glass/glass.data"
    colunas = ['Id', 'RI', 'Na', 'Mg', 'Al', 'Si', 'K', 'Ca', 'Ba', 'Fe', 'Class']

        df = pd.read_csv(url, names=colunas, header=None)

        y = df['Class'].values
        return df, X, y

def plotar_radar_grid(X_df, y_raw):
    # Normalização Min-Max para visualização
    X_norm = (X_df - X_df.min()) / (X_df.max() - X_df.min())
    X_norm['Class'] = y_raw
    df_medias = X_norm.groupby('Class').mean()
    
    classes_presentes = df_medias.index.tolist()
    cols, rows = 3, 2 
    fig, axs = plt.subplots(nrows=rows, ncols=cols, subplot_kw=dict(polar=True), figsize=(15, 10))
    axs = axs.flatten()
    
    categorias = list(df_medias.columns)
    N = len(categorias)
    angulos = [n / float(N) * 2 * pi for n in range(N)]
    angulos += angulos[:1] 
    
    nome_classes_map = {1: '1: Janela Float', 2: '2: Janela N-Float', 3: '3: Veículo', 5: '5: Container', 6: '6: Louça', 7: '7: Farol'}
    
    lista_cores = ['tab:blue', 'gold', 'tab:green', 'tab:red', 'tab:purple', 'tab:pink']
    
    for i, classe_idx in enumerate(classes_presentes):
        ax = axs[i]
        valores = df_medias.loc[classe_idx].values.flatten().tolist()
        valores += valores[:1] 
        
        ax.set_xticks(angulos[:-1])
        ax.set_xticklabels(categorias, color='grey', size=8)
        ax.set_ylim(0, 1)
        ax.set_yticks([0.25, 0.5, 0.75])
        ax.set_yticklabels([]) 
        
        cor_atual = lista_cores[i % len(lista_cores)]
        ax.plot(angulos, valores, linewidth=2, linestyle='solid', color=cor_atual)
        ax.fill(angulos, valores, color=cor_atual, alpha=0.4) 
        ax.set_title(nome_classes_map.get(classe_idx), size=14, weight='bold', color='black', pad=20)

    plt.suptitle("DNA Químico: Comparação Visual das Classes de Vidro", fontsize=20, y=1.02)
    plt.tight_layout()
    plt.show()


# 03: rede neural

def sigmoid(z): 
    z = np.clip(z, -500, 500)
    return 1 / (1 + np.exp(-z))

def sigmoid_derivada(a): return a * (1 - a)

def softmax(z):
    exp_z = np.exp(z - np.max(z, axis=1, keepdims=True))
    return exp_z / np.sum(exp_z, axis=1, keepdims=True)

class RedeNeuralVidros:
    def __init__(self, n_entradas, n_ocultas, n_saidas, regularizacao_lambda): 
        np.random.seed(42)
        self.regularizacao_lambda = regularizacao_lambda 
        self.W1 = np.random.randn(n_entradas, n_ocultas) * np.sqrt(1. / n_entradas)
        self.b1 = np.zeros((1, n_ocultas))
        self.W2 = np.random.randn(n_ocultas, n_saidas) * np.sqrt(1. / n_ocultas)
        self.b2 = np.zeros((1, n_saidas))
        self.loss_treino = []
        self.loss_teste = [] 

    def forward(self, X):
        self.z1 = np.dot(X, self.W1) + self.b1
        self.a1 = sigmoid(self.z1)
        self.z2 = np.dot(self.a1, self.W2) + self.b2
        self.a2 = softmax(self.z2)
        return self.a2

    def calcular_perda(self, y_true, y_pred):
        m = y_true.shape[0]
        log_likelihood = -np.log(y_pred[range(m), y_true.argmax(axis=1)] + 1e-9)
        loss_principal = np.sum(log_likelihood) / m
        penalidade_l2 = (self.regularizacao_lambda / (2 * m)) * (np.sum(self.W1**2) + np.sum(self.W2**2))
        return loss_principal + penalidade_l2

    def backward(self, X, y_true, learning_rate):
        m = X.shape[0]
        dz2 = self.a2 - y_true
        gradiente_w2 = (np.dot(self.a1.T, dz2) / m) + (self.regularizacao_lambda / m) * self.W2
        db2 = np.sum(dz2, axis=0, keepdims=True) / m
        da1 = np.dot(dz2, self.W2.T)
        dz1 = da1 * sigmoid_derivada(self.a1)
        gradiente_w1 = (np.dot(X.T, dz1) / m) + (self.regularizacao_lambda / m) * self.W1
        db1 = np.sum(dz1, axis=0, keepdims=True) / m
        self.W1 -= learning_rate * gradiente_w1
        self.b1 -= learning_rate * db1
        self.W2 -= learning_rate * gradiente_w2
        self.b2 -= learning_rate * db2

    def treinar(self, X_treino, y_treino, X_teste, y_teste, epocas, learning_rate):
        print(f"Iniciando treinamento (Lambda={self.regularizacao_lambda})...")
        for i in range(epocas):
            y_pred_treino = self.forward(X_treino)
            loss_t = self.calcular_perda(y_treino, y_pred_treino)
            self.loss_treino.append(loss_t)
            self.backward(X_treino, y_treino, learning_rate)
            y_pred_teste = self.forward(X_teste)
            loss_val = self.calcular_perda(y_teste, y_pred_teste)
            self.loss_teste.append(loss_val)
            if i % 100 == 0:
                print(f"Época {i}: Treino Loss={loss_t:.4f} | Teste Loss={loss_val:.4f}")

    def prever(self, X):
        y_pred = self.forward(X)
        return np.argmax(y_pred, axis=1)

# 04: preparação dos dados

def preparar_dados_rna(df, y_raw):

    X = df.drop(['Id', 'Class'], axis=1).values
    y = df['Class'].values
    
    # Normalização
    media = np.mean(X, axis=0)
    desvio = np.std(X, axis=0)
    X_scaled = (X - media) / (desvio + 1e-8)
    
    mapa_classes = {1:0, 2:1, 3:2, 5:3, 6:4, 7:5}
    y_indices = np.array([mapa_classes[val] for val in y])
    n_classes = 6
    n_samples = len(y_indices)
    
    # One-Hot Encoding
    y_onehot = np.zeros((n_samples, n_classes))
    y_onehot[np.arange(n_samples), y_indices] = 1
    
    # 70% treino // 30% teste
    np.random.seed(42)
    indice_embaralhado = np.random.permutation(n_samples)
    X_shuffled = X_scaled[indice_embaralhado]
    y_shuffled = y_onehot[indice_embaralhado]
    
    ponto_corte = int(0.7 * n_samples)
    return X_shuffled[:ponto_corte], X_shuffled[ponto_corte:], y_shuffled[:ponto_corte], y_shuffled[ponto_corte:], n_classes

# 05: métricas do modelo

def calcular_matriz_confusao_manual(y_true, y_pred, n_classes):
    matriz = np.zeros((n_classes, n_classes), dtype=int)
    for i in range(len(y_true)):
        matriz[y_true[i]][y_pred[i]] += 1
    return matriz

def gerar_relatorio_metricas(y_true, y_pred, mapa_nomes):

    classes_unicas = sorted(list(set(y_true) | set(y_pred)))
    
    dados_tabela = []
    colunas_tabela = ["Classe", "Precisão", "Recall", "F1-Score"]
    
    medias_precisao = []
    medias_recall = []
    medias_f1 = []
    
    # Variáveis para média ponderada
    soma_precisao_pond = 0
    soma_recall_pond = 0
    soma_f1_pond = 0
    total_amostras = 0
    
    for c in classes_unicas:
        tp = np.sum((y_true == c) & (y_pred == c))
        fp = np.sum((y_true != c) & (y_pred == c))
        fn = np.sum((y_true == c) & (y_pred != c))
        
        # Suporte real
        support = np.sum(y_true == c)
        total_amostras += support
        
        precisao = tp / (tp + fp) if (tp + fp) > 0 else 0.0
        recall = tp / (tp + fn) if (tp + fn) > 0 else 0.0
        f1 = 2 * (precisao * recall) / (precisao + recall) if (precisao + recall) > 0 else 0.0
        
        medias_precisao.append(precisao)
        medias_recall.append(recall)
        medias_f1.append(f1)
        
        soma_precisao_pond += precisao * support
        soma_recall_pond += recall * support
        soma_f1_pond += f1 * support
        
        nome_classe = mapa_nomes.get(c, str(c))
        dados_tabela.append([nome_classe, f"{precisao*100:.2f}%", f"{recall*100:.2f}%", f"{f1*100:.2f}%"])
        
    if medias_f1:
        # Macro
        media_p_macro = np.mean(medias_precisao) * 100
        media_r_macro = np.mean(medias_recall) * 100
        media_f_macro = np.mean(medias_f1) * 100
        
        # Ponderada
        media_p_pond = (soma_precisao_pond / total_amostras) * 100 if total_amostras > 0 else 0
        media_r_pond = (soma_recall_pond / total_amostras) * 100 if total_amostras > 0 else 0
        media_f_pond = (soma_f1_pond / total_amostras) * 100 if total_amostras > 0 else 0

        dados_tabela.append(["MÉDIA MACRO", f"{media_p_macro:.2f}%", f"{media_r_macro:.2f}%", f"{media_f_macro:.2f}%"])
        dados_tabela.append(["MÉDIA PONDERADA", f"{media_p_pond:.2f}%", f"{media_r_pond:.2f}%", f"{media_f_pond:.2f}%"])

    plt.figure(figsize=(10, len(dados_tabela)*0.6 + 1.5))
    ax = plt.gca()
    ax.axis('off')
    
    tabela = ax.table(
        cellText=dados_tabela,
        colLabels=colunas_tabela,
        loc='center',
        cellLoc='center',
        colColours=['#40466e']*4
    )
    
    tabela.auto_set_font_size(False)
    tabela.set_fontsize(12)
    tabela.scale(1, 1.5)
    
    for (row, col), cell in tabela.get_celld().items():
        if row == 0: 
            cell.set_text_props(weight='bold', color='white')
        else:
            if row >= len(dados_tabela) - 1: 
                cell.set_text_props(weight='bold')
                cell.set_facecolor('#d9d9d9') 
            elif row % 2 == 0:
                cell.set_facecolor('#f2f2f2') 

    plt.title("Relatório de Métricas por Classe", fontsize=16, weight='bold', pad=20)
    plt.show()
